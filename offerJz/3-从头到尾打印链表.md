# 从头到尾打印链表

```java
import java.util.*;
public class Solution {
    // 1. 栈
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode) {
      
       ArrayList<Integer> result = new ArrayList<Integer>();
	  Stack<Integer> ss = new Stack<Integer>();
	  ListNode startNode = listNode;
     
	  while(listNode!= null){
		  ss.push(listNode.val);
		  listNode = listNode.next;
	  }
	  while(ss.isEmpty()!=true){
		  result.add(ss.pop());
	  }
	  return result;
    }
    // 2. 递归 本质也是调用栈 系统栈
    public ArrayList<Integer> printListFromTailToHead(ListNode){
        ArrayList<Integer> result = new ArrayList<Integer>();
        return myStack(listNode,result);
    }
    public ArrayList<Integer> myStack( ListNode listNode,ArrayList<Integer> result){
         if(listNode != null){
             myStack(listNode.next,result);
             result.add(listNode.val);
         }
         return result;
    }
}
```
