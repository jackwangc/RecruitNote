# 必备算法

## 归并排序

```java
    public void mergeSort(int[] arr){
        // 1. 数组长度
        int len = arr.length;
        // 2. 分枝
        sort(arr, 0, len - 1);     
    }
    private void sort(int[] arr, int left, int right){
        if (left < right) {
            int mid = left + (right - left) >> 1;
            // 左边数组归并
            sort(arr, left, mid);
            // 右边数组归并
            sort(arr, mid + 1, right);
            // 3. 合并         
            merge(arr, left, mid, right);
        }
    }
    public static void merge(int[] data, int left, int center, int right) {
        // 临时数组
        int[] tmpArr = new int[data.length];
        // 右数组第一个元素索引
        int mid = center + 1;
        // third 记录临时数组的索引
        int third = left;
        // 缓存左数组第一个元素的索引
        int tmp = left;
        while (left <= center && mid <= right) {
        	// 从两个数组中取出最小的放入临时数组
            if (data[left] <= data[mid]) {
                tmpArr[third++] = data[left++];
            } else {
                tmpArr[third++] = data[mid++];
            }
        }
        // 剩余部分依次放入临时数组(实际上两个 while 只会执行其中一个)
        while (mid <= right) {
            tmpArr[third++] = data[mid++];
        }
        while (left <= center) {
            tmpArr[third++] = data[left++];
        }
        // 将临时数组中的内容拷贝回原数组中
        // (原 left-right 范围的内容被复制回原数组)
        while (tmp <= right) {
            data[tmp] = tmpArr[tmp++];
        }       
    }
```

## 快速排序

```java
    // 快速排序
    public int[] quickSort(int[] arr) {
        return quickSort(arr, 0, arr.length - 1);  
    }
    // 递归
    private int[] quickSort(int[] arr, int left, int right){
        if (left < right){
            // 寻找基准值的交换位置
            int partitionIndex = partition(arr, left, right);  
            quickSort(arr, left, partitionIndex - 1);
            quickSort(arr, partition + 1, right);
        }
        return arr;
    }
    // 寻找基准值 双边扫描
    private int partition(int[] arr, int left, int right){
        int i = left, j = right;
        int temp = arr[left];
        while(i != j){
            while(i < j && arr[i] < temp){
                i++;
            }
            while(j > i && arr[j] > temp){
                j--;
            }
            if (i < j){
                swap(arr, i, j);
            }
        }        
        arr[left] = arr[i]
        arr[i] = temp;
        return i;
    }
    // 单边扫描，双指针
    private int partition(int[] arr, int left, int right){
        int pivot = left;
        int index = left + 1;
        for(int i = pivot; i <= right; i++){
            if(arr[i] < arr[pivot]){
                swap(arr, i, index);
                index++;
            }
        }
        swap(arr, pivot, index - 1);
        return index - 1;
    }
```

## 堆排序

```java
    public void heapSort(int[] arr){
        int len = arr.length;
        // 1. 构建大顶堆
        // 从第一个非叶子结点开始，从左至右，从上到下
        for (int i = len / 2 - 1; i >= 0; i--){
            adjustHeap(arr, i, len -1);
        }
        // 2. 调整
        for (int i = len - 1; i >= 0; i--){
            // 取出最大元素
            swap(arr, 0, i);
            // 重新构建大顶堆
            adjustHeap(arr, 0, i - 1);
        }

    }
    public static void adjustHeap(int []arr,int i,int length){
        int temp = arr[i];//先取出当前元素i
        for(int k=i*2+1;k<length;k=k*2+1){//从i结点的左子结点开始，也就是2i+1处开始
            if(k+1<length && arr[k]<arr[k+1]){//如果左子结点小于右子结点，k指向右子结点
                k++;
            }
            if(arr[k] >temp){//如果子节点大于父节点，将子节点值赋给父节点（不用进行交换）
                arr[i] = arr[k];
                i = k;
            }else{
                break;
            }
        }
        arr[i] = temp;//将temp值放到最终的位置
    }
```

## 插入排序

```java
    private int[] selectSort(int[] arr){
        int len = arr.len;
        for(int i = 1; i < len ; i++){
            int temp = arr[i];
            int j = i;
            while(j > 0 && temp < arr[j - 1]){
                arr[j] = arr[j - 1];
                j--;
            }
            if(j != i){
                arr[j] = temp;
            }
        }
        return arr;
    }
```

## 二分查找

```java
    public void solution(int[] arr, int key) {
        int high = arr.length - 1;
        int low = 0;
        while (low <= high) {
            int mid = low + (high - low) >> 1;
            if (arr[mid] > key) {
                high = mid - 1;
            } else if (arr[mid] < key){
                low = mid + 1;
            } else {
                return mid;
            }
        }
    }
```

## 二叉树先序遍历

```java
    // 1. 递归写法
    public void recursionPreorderTraversal(TreeNode root) {
        if (root != null) {
            // 根左右
            // 1. 找到打印出根的值
            System.out.println(root.val);
            // 2. 找到左节点 对其进行根左右 不用多余判断左右节点是否为空 下一个循环会自动判断
            recursionPreorderTraversal(root.left);
            // 3. 找到右节点，对其进行根左右
            recursionPreorderTraversal(root.right);
        }
    }
    // 2. 非递归写法
    public void recursionPreorderTraversal2(TreeNode root) {
        if (root == null){
            return;
        }
        Stack<TreeNode> stack = new Stack<>();
        TreeNode node = root;
        while (root != null || stack.isEmpty()){
            while (node != null){
                stack.push(node);
                // 进行操作
                node = node.left;
            }
            if (!stack.isEmpty()){
                node = stack.pop();
                node = node.right;
            }
        }
    }
```

## 二叉树中序遍历

```java
    public static void inorderTraversalRec(TreeNode root){
        if (root == null) {
            return;
        }
        inorderTraversalRec(root.left);
        System.out.print(root.val + "->");
        inorderTraversalRec(root.right);
    }

    public static void inorderTraversal(TreeNode root) {
        if (root == null) {
            return;
        }
        Stack<TreeNode> stack = new Stack<>(); // 辅助栈
        TreeNode cur = root;
        while (cur != null || !stack.isEmpty()) {
            while (cur != null) { // 不断将左子节点入栈，直到cur为空
                stack.push(cur);
                cur = cur.left;
            }
            if (!stack.isEmpty()) { // 栈不为空，弹出栈元素
                cur = stack.pop(); // 此时弹出最左边的节点
                System.out.print(cur.val + "->"); // 中序遍历，先打印左子节点在打印当前节点，然后再把右子节点加到栈中
                cur = cur.right; // 令当前节点为右子节点
            }
        }
    }
```

## 二叉树后序遍历

```java
    public static void postorderTraversalRec(TreeNode root){
        if (root == null) {
            return;
        }
        postorderTraversalRec(root.left);
        postorderTraversalRec(root.right);
        System.out.print(root.val + "->");
    }
    public static void postorderTraversal(TreeNode root) {
        if (root != null){
            return;
        }
        Stack<TreeNode> stack = new Stack();
        TreeNode node = root;
        TreeNode LastNode = root;
        while (node != null || stack.isEmpty()){
            while (node != null){
                stack.push(node);
                node = node.left;
            }
            node = stack.peek();
            if (node.right == null || node.right == lastNode) {
                // 操作
                stack.pop();
                lastNode = node;
                node == null;
            }else{
                node = node.right;
            }
        }
    }
```

```java
    public static void levelTraversal(TreeNode root){
        if(root == null) {
            return;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);

        while (!queue.isEmpty()){
            // 利用队列先进先出 广度优先搜索
            TreeNode temp= queue.poll();  // 删除第一个元素并从队列中删除
            System.out.print("");
            if (temp.left!=null){
                queue.add(temp.left);
            }
            if (temp.right!=null){
                queue.add(temp.right);
            }
        }
    }
```

```java
    class God{
        private static final God god = new God();

    // 懒汉模式
    class God{
        private volatile static God god;
        private God(){};
        public static God getInstance(){
            if (god == null){
                synchronized(God.class){
                    god = new God();
                }
            }
        }
    }
```
